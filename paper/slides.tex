%% --- equal-height card rows -------------------------------------------------
%% \eqcards{ \rc{width}{overlay}{cardcode} ...  \rcx{width}{plain} ... }
%% Measures every \rc card at its natural height, then lays them out in a
%% [T] columns row forcing all to the tallest, so tops AND bottoms align.
%% Each card must take its height from [\eqrowht], e.g.
%%   \rc{0.48\textwidth}{2-}{\begin{dangercard}[\eqrowht]{Title}...\end{dangercard}}
%% \rcx is an un-measured column (arrows, spacers). Overlay "1-" = always shown.
\makeatletter
\newlength{\eq@h}
\newcommand{\eqrowht}{}
\long\def\eqcards#1{%
  \setlength{\eq@h}{\z@}\let\eqrowht\cardnil
  \begingroup
  \long\def\rc##1##2##3{\setbox\@tempboxa=\hbox{\begin{minipage}{##1}##3\end{minipage}}%
    \dimen@=\ht\@tempboxa\advance\dimen@\dp\@tempboxa
    \ifdim\dimen@>\eq@h\global\eq@h=\dimen@\fi}%
  \long\def\rcx##1##2{}%
  #1%
  \endgroup
  \advance\eq@h by-8\p@ \edef\eqrowht{\the\eq@h}%
  \begin{columns}[T,onlytextwidth]%
    \long\def\rc##1##2##3{\column{##1}\onslide<##2>{##3}}%
    \long\def\rcx##1##2{\column{##1}##2}%
    #1%
  \end{columns}}
%% \eqmeasure{ \mc{width}{cardcode} ... } sets \eqrowht to the tallest card
%% (cards use [\eqrowht]); lay the cards out yourself afterwards. Useful for
%% grids where the same height must span two stacked rows.
\long\def\eqmeasure#1{%
  \setlength{\eq@h}{\z@}\let\eqrowht\cardnil
  \begingroup
  \long\def\mc##1##2{\setbox\@tempboxa=\hbox{\begin{minipage}{##1}##2\end{minipage}}%
    \dimen@=\ht\@tempboxa\advance\dimen@\dp\@tempboxa
    \ifdim\dimen@>\eq@h\global\eq@h=\dimen@\fi}%
  #1%
  \endgroup
  \advance\eq@h by-8\p@ \edef\eqrowht{\the\eq@h}}
%% \eqhalfheight{width}{content} measures a tall box and stores in \rightht the
%% content height for each of TWO stacked cards so together they equal it. Used
%% on slide 18 (left REPL box == the two right cards combined).
\newcommand{\rightht}{}
\newcommand{\eqhalfheight}[2]{%
  \setbox\@tempboxa=\hbox{\begin{minipage}{#1}#2\end{minipage}}%
  \@tempdima=\ht\@tempboxa\advance\@tempdima\dp\@tempboxa
  \edef\rightht{\the\dimexpr0.5\@tempdima-11pt\relax}}
%% \eqmatchcol{rightW}{rightContent}{leftW}{leftCard} makes a left card's TOTAL
%% height exactly equal the stacked right column, with no guessing about the
%% card's own padding. \leftCard must be a macro that draws the left card using
%% [\lefthgt] for its height. Two passes: (1) measure right total R and the left
%% card's total when its content height is set to R, giving overhead = L-R; then
%% (2) set \lefthgt = R - overhead = 2R - L, so the laid-out left total is R.
\newcommand{\lefthgt}{}
\newcommand{\eqmatchcol}[4]{%
  \setbox\@tempboxa=\hbox{\begin{minipage}[t]{#1}#2\end{minipage}}%
  \@tempdima=\dimexpr\ht\@tempboxa+\dp\@tempboxa\relax % R = right total
  \edef\lefthgt{\the\@tempdima}%
  \setbox\@tempboxa=\hbox{\begin{minipage}[t]{#3}#4\end{minipage}}%
  \@tempdimb=\dimexpr\ht\@tempboxa+\dp\@tempboxa\relax % L = left total at height R
  \edef\lefthgt{\the\dimexpr2\@tempdima-\@tempdimb\relax}}
%% \eqgrid{colwidth}{ovA}{cardA}{ovB}{cardB}{ovC}{cardC}{ovD}{cardD}
%% 2x2 grid: A,B stacked in the left column; C,D in the right. Each ROW's two
%% cards are forced to a common height (A=C, B=D), so cells align across columns
%% and rows without inflating the whole grid to one global maximum. Cards read
%% their height from \eqrowht, which is switched per row during layout.
\newcommand{\eqrha}{}\newcommand{\eqrhb}{}
\long\def\eq@rowht#1#2#3{% #1=width #2,#3 the two cards; result in \@tempdima
  \setbox\@tempboxa=\hbox{\begin{minipage}{#1}#2\end{minipage}}%
  \@tempdima=\ht\@tempboxa\advance\@tempdima\dp\@tempboxa
  \setbox\@tempboxa=\hbox{\begin{minipage}{#1}#3\end{minipage}}%
  \dimen@=\ht\@tempboxa\advance\dimen@\dp\@tempboxa
  \ifdim\dimen@>\@tempdima\@tempdima=\dimen@\fi}
\long\def\eqgrid#1#2#3#4#5#6#7#8#9{%
  \let\eqrowht\cardnil
  \eq@rowht{#1}{#3}{#7}\edef\eqrha{\the\dimexpr\@tempdima-8pt\relax}%
  \eq@rowht{#1}{#5}{#9}\edef\eqrhb{\the\dimexpr\@tempdima-8pt\relax}%
  \begin{columns}[T,onlytextwidth]%
    \column{#1}\let\eqrowht\eqrha\onslide<#2>{#3}\smallskip\par
               \let\eqrowht\eqrhb\onslide<#4>{#5}%
    \column{#1}\let\eqrowht\eqrha\onslide<#6>{#7}\smallskip\par
               \let\eqrowht\eqrhb\onslide<#8>{#9}%
  \end{columns}}
\makeatother

%% --- overlay-aware speaker-note beats ---------------------------------------
%% Wrap a note in \notescript{...}; write each spoken beat as \beat{k}{text}
%% (or \lbeat{k}{text} for the final one). \notescript emits one note page per
%% overlay, highlighting the beat for that overlay and dimming the rest, so you
%% can see where you are. It does NOT add overlays: each \note<i> is accumulated
%% at the right overlay during frame processing (beamer's slide-in-frame counter
%% is unreliable inside notes because the note is typeset at the next slide).
%% highlight in srcBlue (the "Static guarantee" card blue): strong contrast
%% against the dimmed gray beats and easy to read.
\newcommand{\beathl}[1]{{\color{srcBlue}#1}}
\newcommand{\beatdim}[1]{{\color{black!48}#1}}
%% click icon shown before every beat except the first (beat 1 needs no click),
%% so the number of clicks in a slide is visible at a glance.
\newcommand{\nbicon}{\faMousePointer}
\makeatletter
\newcount\nb@total \newcount\nb@i \newcount\nb@k
%% runtime renderer for one beat: advance the running index, highlight it if it
%% matches the (baked) active overlay, else dim. \nb@active is set per note page.
\newcommand{\nb@item}[1]{\global\advance\nb@k by1
  \ifnum\nb@k>1 \def\nb@ic{\nbicon~}\else\def\nb@ic{}\fi
  \ifnum\nb@k=\nb@active\relax\beathl{\nb@ic#1}\else\beatdim{\nb@ic#1}\fi\par\smallskip}
\long\def\notescript#1{%
  \global\nb@total=0 \gdef\nb@list{}\gdef\nb@header{}%
  %% pass 1: capture header + each beat's text (typesetting absorbed by the box)
  \setbox\z@\vbox{\hsize\textwidth
    \long\def\textbf##1{\gdef\nb@header{\textbf{##1}}}%
    \long\def\beat##1##2{\global\advance\nb@total by1
      \g@addto@macro\nb@list{\nb@item{##2}}}%
    \let\lbeat\beat
    \def\\{}\def\quad{}%
    #1}%
  \ifnum\nb@total=0\relax
    \note{#1}%  no beats: emit a plain note unchanged
  \else
    %% emit one \note<i> per overlay; bake header, active index, and beat list
    %% fully into each note (via \unexpanded) so late typesetting stays correct.
    \nb@i=0
    \loop\ifnum\nb@i<\nb@total
      \advance\nb@i by1
      \edef\nb@do{%
        \noexpand\note<\the\nb@i>{%
          \unexpanded\expandafter{\nb@header}%
          \noexpand\par\noexpand\smallskip
          \noexpand\global\noexpand\nb@k=0\relax
          \noexpand\def\noexpand\nb@active{\the\nb@i}%
          \unexpanded\expandafter{\nb@list}}}%
      \nb@do
    \repeat
  \fi}
\makeatother

\begin{frame}{\texttt{sprintf}: a format value can determine a function's type}
  \begin{center}
  \begin{tikzpicture}[node distance=0.55cm and 2.15cm]

      (typ1) {{\large\texttt{Nat -> Str}}};
    \node[source node,below=of fmt1,minimum width=2.2cm,minimum height=0.9cm]
      (fmt2) {{\Large\texttt{"\%d\%s"}}};
    \node[target node,right=of fmt2,minimum width=4.0cm,minimum height=0.9cm]
      (typ2) {{\large\texttt{Nat -> Str -> Str}}};
    \draw[flow arrow,srcBlue] (fmt1) -- node[above,black!55,font=\scriptsize]
      {determines} (typ1);
    \draw[flow arrow,srcBlue] (fmt2) -- node[above,black!55,font=\scriptsize]
      {determines} (typ2);
  \end{tikzpicture}
  \end{center}

  \pause
  \medskip
  \begin{sourcecard}{\faProjectDiagram\quad The dependent-type idea}
    A \emph{value} --- the format --- determines the \emph{type} of the
    function that interprets it.
  \end{sourcecard}
  \notescript{%
    \textbf{[0:30 — 1:30]}\\
    \beat{1}{Here is the running example, and you already know the function:
    sprintf. The format percent-d describes a function that consumes a natural
    number and returns a string. Percent-d percent-s describes a function with
    two arguments. The format is ordinary data, but it determines the
    function's type.}\\
    \lbeat{2}{That is what ``dependent'' means here: a value appears in, or
    computes, a type. The source type checker can therefore keep the format
    and every call site synchronized. Vectors, matrices, and the rest later
    are the same idea with a different value driving the type; you do not need
    the Calculus of Constructions to follow any of it.}
  }
\end{frame}

\begin{frame}[fragile]{Enjoy the dependent guarantees with no limits}
  %% \rrh = each right box's height. \eqmatchcol makes the left box's TOTAL height
  %% exactly equal the two stacked right boxes. Inside the left box: a little space

      \textcolor{blameRed}{blame}.
    \end{dynamiccard}}
    \rc{0.31\textwidth}{3-}{\begin{dangercard}[\eqrowht]{\faTimesCircle\quad Never}
      Hide the gap behind an unchecked cast and undefined behavior.
    \end{dangercard}}
  }

    Can we verify both the translation and its fallback?
  \end{alertblock}}
  \notescript{%
    \textbf{[4:50 — 5:50]}\\
    \beat{1}{This gives us a hierarchy of outcomes. First, the best outcome is

    \beat{2}{If the target genuinely cannot express the dependency, use a typed,

    \beat{3}{What we must not do is pretend the types agree and delegate the

\begin{frame}{The type-theoretic gap, precisely}

  \begin{dynamiccard}{\faBalanceScale\quad The middle ground}

    Calculus of Constructions, the small dependent core behind Rocq. Its central
    type is the dependent product, Pi x of A, B x. Ordinary functions and
    universal quantification are both instances of it, and the result type may
    mention the value x. CoC also treats terms and types in one uniform syntax.}\\

    substantially richer than ML polymorphism: it has explicit type abstraction
    and application, type-level lambda functions, and higher kinds. That is why
    it can preserve structures like Fmt and computed type-level codes. But it
    still separates terms from types, so an ordinary term value cannot occur in a
    target type.}\\
    \lbeat{3}{That makes F-omega a useful middle ground. Paulin-Mohring already
    connected CoC and F-omega in 1989. We keep that expressive static skeleton
    and add Dyn and blame for the part of dependency that crosses the remaining
    term-to-type boundary.}
  }
\end{frame}

    \textbf{[13:35 — 14:50]}\\
    \beat{1}{The first implementation difficulty is the namespace mismatch we
    just saw. CoC deliberately treats programs and types uniformly. These two
    binders have the same syntactic form: one binds a type A and one binds a
    value n.}\\
    \beat{2}{F-omega separates those worlds. A becomes a type abstraction, while
    n becomes an ordinary function argument. The translation must also compress
    one source de Bruijn namespace into separate target type and term

    syntax, because an annotation may compute before revealing what it is. We
    use the semantic question ``does this have sort Kind?'' The source
    normalization and subject-reduction results are what let us prove that this
    classification does not change under computation.}

\begin{frame}{What this is --- and is not}
  \eqgrid{0.48\textwidth}%
    {1-}{\begin{neutralcard}[\eqrowht]{\faCode\quad Mathematical translation}
      \small Total and verified, but normalizes types and invokes inference:
      not a production back end.
    \end{neutralcard}}%
    {3-}{\begin{neutralcard}[\eqrowht]{\faProjectDiagram\quad Syntactic simulation}
      \small Not contextual equivalence or a full compiler-correctness theorem.
    \end{neutralcard}}%
    {2-}{\begin{dangercard}[\eqrowht]{\faExclamationTriangle\quad Reachable internal blame}
      \small A type that depends on a runtime value can deliberately fail.
    \end{dangercard}}%
    {4-}{\begin{dynamiccard}[\eqrowht]{\faLockOpen\quad Executable boundary}
      \small No proof erasure or native inductives; opaque proof bodies are accessed.
    \end{dynamiccard}}

  \onslide<5->{%

  \end{center}}
  \notescript{%
    \textbf{[20:20 — 21:35]}\\
    \beat{1}{The result has four important limits. First, the translation is a
    total mathematical function on typing derivations, but it normalizes types
    and calls a decision procedure. It is not an efficient production back end,
    even though the REPL lets us run examples.}\\
    \beat{2}{Second, internal blame is intentionally reachable. It marks a source
    type that depends on runtime data in a way F-omega cannot represent.}\\
    \beat{3}{Third, simulation is syntactic. We do not yet prove that arbitrary
    typed program contexts observe equivalent behavior; that needs a new typed
    logical relation.}\\
    \beat{4}{Fourth, the current translation does not erase proof terms, the bare
    CoC source models inductive families rather than adding native CIC
    inductives, and Rocq extraction of the executable accesses opaque
    computational proof bodies.}\\
    \lbeat{5}{These limitations do not weaken the stated theorems. They say exactly
    what stage of the design space this artifact establishes.}
  }
\end{frame}
